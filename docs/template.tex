% ---- headings (body) -------------------------------------------------------
% Each top-level section (chapter) starts on a fresh page. We clearpage
% BEFORE \section runs so the TOC records the new page number, not the old.

\renewcommand{\section}{\clearpage\piPaletteOrigSection}

% ---- aperture mark (mirrors the SVG in templates/partials/topbar.html) -----
\newcommand{\apertureMark}[1]{%
  \begin{tikzpicture}[line cap=round]
    \draw[accent, line width=#1] (0,0) circle (2.4cm);
    \draw[accent, line width=#1] (0,0) circle (1.2cm);
    \draw[accent, line width=#1] (0, 2.4) -- (0, 1.6);
    \draw[accent, line width=#1] (0,-2.4) -- (0,-1.6);
    \draw[accent, line width=#1] (-2.4,0) -- (-1.6,0);
    \draw[accent, line width=#1] ( 2.4,0) -- ( 1.6,0);
  \end{tikzpicture}%
}

  \end{titlepage}
